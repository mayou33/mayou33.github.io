     关闭防火墙

    #停止firewall

/etc/init.d/iptables stop

&nbsp;#禁止firewall开机启动

systemctl stop firewalld.service

systemctl disable firewalld.service

关闭selinux

setenforce 0

echo &quot;setenforce 0&quot; &gt;&gt; /etc/rc.local

sed -i &#39;s/SELINUX=enforcing/SELINUX=disabled/g&#39;&nbsp;&nbsp;&nbsp;&nbsp; /etc/sysconfig/selinux

不能用centos7 ，php版本支持到5.4。

[https://github.com/mfreiholz/iF.SVNAdmin](https://github.com/mfreiholz/iF.SVNAdmin)

git clone [https://github.com/mfreiholz/iF.SVNAdmin.git](https://github.com/mfreiholz/iF.SVNAdmin.git)

yum -y install httpd httpd-devel php

yum -y install subversion mod_dav_svn

cp -rf iF.SVNAdmin /var/www/html/

touch /data/svn/accessfile

chown -R apache.apache /data/svn/
