###Percona版 —-推荐

yum -y install https://www.percona.com/redir/downloads/percona-release/redhat/latest/percona-release-0.1-6.noarch.rpm

yum -y install Percona-Server-client-57 Percona-Server-devel-57 Percona-Server-shared-57

######yum -y install Percona-Server-client-56 Percona-Server-devel-56 Percona-Server-shared-56

#rpm -ql Percona-Server-client-57

#####mysql 社区版

rpm -ivh https://dev.mysql.com/get/mysql57-community-release-el7-11.noarch.rpm

yum install -y mysql-community-client

#####mariadb版

rpm –import https://yum.mariadb.org/RPM-GPG-KEY-MariaDB

yum install MariaDB-client -y

#rpm -ql MariaDB-client
